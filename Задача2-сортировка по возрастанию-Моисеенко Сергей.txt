//Функция сортировки нового массива по возрастанию
void fun(int num1, int n, int *arr1, int *arr2, int *arr3){
    int f=0, k=0;//Переменные функции
    for (int i=0;i<n;i++){
        if (i<=num1-1){
            arr3[i]=arr1[i];
        }
        else if (i>num1-1){
            arr3[i]=arr2[f];
            f++;
        }
    }//Объединение первого и второго массива
    for (int i=0;i<n-1;i++){
        for (int j=0;j<n-i-1;j++){
            if (arr3[j]>arr3[j+1]){
                k=arr3[j];
                arr3[j]=arr3[j+1];
                arr3[j+1]=k;
            }
        }
    }//Пузырчатая сортировка нового массива
}

int main()
{
    int num1=4, num2=4,n=num1+num2;//Размерность массивов
    int arr1[num1]={1,4,8,11};//Первый отсортированный по возрастанию массив
    int arr2[num2]={-6,6,8,9};//Второй отсортированный по возрастанию массив
    int arr3[n];//Новый массив по размеру равный первым двум
    fun(num1, n, arr1, arr2, arr3);//Функция по сортировке нового массива
    for (int i=0;i<n;i++){
        cout<<arr3[i]<<' ';
    }//Вывод отсортированного нового массива
}
