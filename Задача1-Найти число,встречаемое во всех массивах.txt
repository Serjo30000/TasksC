void min(int arri1,int arrj2,int arrk3, int &i, int &j, int &k){//Функция нахождения минимального элемента в массиве для перехода к следующему элементу
    if (arri1<arrj2){//Переход к следующему элементу у первого массива
        i++;
    }
    else if (arrj2<arrk3){//Переход к следующему элементу у второго массива
        j++;
    }
    else{//Переход к следующему элементу у третьего массива
        k++;
    }
}

void fun(bool &f,int &res,int size1, int size2, int size3, int *arr1, int *arr2, int *arr3){//Функция нахождения числа, имеющегося в трёх массивах
    int i=0,j=0,k=0;//Счётчики
    while (i<size1 and j<size2 and k<size3){//Повторяющийся цикл нахождения трёх равных элементов массивов
        if (arr1[i]==arr2[j] and arr1[i]==arr3[k]){//Условие нахождения трёх равных элементов
            res=arr1[i];
            f=true;//Имеются три равные элемента
            break;
        }
        min(arr1[i],arr2[j],arr3[k],i,j,k);//Функция нахождения минимального элемента в массиве для перехода к следующему элементу
    }
}

int main()
{
    int arr1[7]={3,4,4,4,5,6,7};//Первый отсортированный массив
    int arr2[9]={1,2,3,4,4,5,6,8,9};//Второй отсортированный массив
    int arr3[8]={5,6,7,8,9,10,11,12};//Третий отсортированный массив
    int size1=0,size2=0,size3=0;//Длины массивов
    int res=0;//Результат
    bool f=false;//Флаг истинности
    size1=sizeof(arr1)/sizeof(int);//Длина первого отсортированного по вохростанию массива
    size2=sizeof(arr2)/sizeof(int);//Длина второго отсортированного по вохростанию массива
    size3=sizeof(arr3)/sizeof(int);//Длина третьего отсортированного по вохростанию массива
    fun(f,res,size1,size2,size3,arr1,arr2,arr3);//Функция нахождения числа, имеющегося в трёх массивах
    if (f==true){
        cout<<res;//Вывод результата
    }
    else{
        cout<<"Нет результата";
    }
}
