#include <iostream>

using namespace std;
//Автор: Моисеенко Сергей Дата: 18.03.2022
int fibonacci(int a){//Функция нахождение числа Фибоначчи
    if (a<=2){//Условие для чисел, которые всегда дают единицу
        return 1;//Вернуть единицу
    }
    int a1=fibonacci(a/2);//Рекурсия нахождение позапрошлого элемента
    int a2=fibonacci(a/2+1);//Рекурсия нахождение прошлого элемента
    if (a%2==0){//Условие для чётного числа
        return a1*(2*a2-a1);//Вернуть значение по формуле для чётного числа
    }
    else{//Условие для нечётного числа
        return a1*a1+a2*a2;//Вернуть значение по формуле для нечётного числа
    }
}

int main()
{
    int k=100;//Количество чисел Фибоначчи
    int n=17711;//Данное число
    for (int i=0;i<=k;i++){//Цикл для сравнения данного числа с числом Фибоначчи
        if (n==fibonacci(i)){//Условие равенства данного числа с числом фибоначчи
            cout<<"Это число "<<n<<" является частью последовательности Фибоначчи";//Вывод о имении последовательности
            break;//Завершить цикл
        }
        else if(fibonacci(i)>n or i==k){//Условие выхода за пределы границ для искомого данного числа в последовательности Фибоначчи
            cout<<"Это число "<<n<<" не является частью последовательности Фибоначчи";//Вывод о неимении последовательности
            break;//Завершить цикл
        }
    }
}
